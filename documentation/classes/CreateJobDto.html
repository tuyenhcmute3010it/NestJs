  IsNotEmpty,
  IsNotEmptyObject,
  IsObject,
  IsString,
  ValidateNested,
} from 'class-validator';
import mongoose from 'mongoose';
class Company {
  @IsNotEmpty()
  _id: mongoose.Schema.Types.ObjectId;

  @IsNotEmpty()
  name: string;
}

  salary: number;

    message: 'quantity Khong duoc de trong',
  })
  quantity: number;

  @IsNotEmpty({
    message: 'level Khong duoc de trong',
  })
  level: string;

  @IsNotEmpty({
    message: 'description Khong duoc de trong',
  })
  description: string;

  @IsNotEmpty({
    message: 'description Khong duoc de trong',
  })
  startDate: Date;

  @IsNotEmpty({
    message: 'description Khong duoc de trong',
  })
  endDate: Date;

  @IsNotEmpty({

}
